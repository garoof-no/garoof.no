## """problem statement"""

So if we're using something like Disjktra's algorithm on something like a tilemap we tend to get one of the minimally expensive paths, according to some way of calculating costs. The mental model could be something like a turn based strategy game: There could be roads and different terrain types having different movement costs and more expensive does not need to mean more moves made or more positions visited.

In these examples things will be fairly simple:

* Terrain is intraversable (`#`) or intraversable (typically space characters)
* Movement cost is 2 for orthogonal movement and 3 for diagonal

Anyway it can be the case that a lot of different paths from one position to another have the same cost. A thing that's been on my mind is something about picking a path that looks kind of "natural." Particularly with stuff like: If we're moving to a position and in order to get there we have to move east 10 times and diagonally to the northeast 3 times, we might get paths like:

```
            o
           o
          o
oooooooooo
``` 10 moves west followed by 3 moves to the northwest

While if we're making the moves by hand we might make a path that looks more like one line (despite both paths being equally good wrt cost). Maybe something like:

```
          ooo
       ooo
   oooo
ooo
``` 3 west, 1 northwest, 4 west, 1 northwest, 3 west, 1 northwest, 3 west

So something to do with that has been on my mind. How would I go about choosing paths that look less "artificial," when the artificial movement cost rules make the "artifical" ones technically as good as the "natural" ones?

## a found path

The data structure we get from the `flood` function also has information about how we got to it. When we look up a position and get a result if it's reachable, the result we get will contain a `from` table, with directions as keys and positions as values. A way to find a path to a position is to start with that position and then step back towards where we started the flood filling by always following the first `from` value we find:

``` run
local function stepback(p, fill)
    local seen = fill[p]
    if not seen then return nil end
    for _, d in ipairs(dirs) do
      if seen.from[d] then return seen.from[d] end
    end
    return nil
end

function test(str)
  local map = mappy(str)
  local start, stop = next(map.S), next(map.E)
  local q = flood(map, start)
  local path = { [stop] = true }
  local p = stepback(stop, q.seen)
  while p do
    path[p] = true
    p = stepback(p, q.seen)
  end
  local function heat(c, p)
    if not path[p] then return c end
    c = c == " " and "o" or c
    return heatcoloured(c, p, q)
  end
  web.html(maphtml(map, heat))
end
```

In _some_ cases it might kind of make more sense to make modifications to the part that searches from the start position instead of to the part that constructs the path from the end position. E.g. a typical hereustic function seems like a more sensible thing to use for guiding the initial search rather than the construction of a path after that, I guess. But in other cases I kind of want to use the completed set of paths _when_ constructing my final path. local function better(a, b)
  for i = 1, math.min(#a, #b) do
    if a[i].count < b[i].count then return true end
    if a[i].count > b[i].count then return false end
  end
  return (#a < #b) and a or b
end

local function best(path, currentbest)
  local prev = path[#path]
  local list, stuf = { prev }, {}
  for i = #path - 1, 1, -1 do
    local p = path[i]
    list[#list + 1] = p
    local dir = prev - p
    if stuf[#stuf] and stuf[#stuf].dir == dir then
      stuf[#stuf].count = stuf[#stuf].count + 1
    else
      stuf[#stuf + 1] = { dir = dir, count = 1 }
    end
    prev = p
  end
  table.sort(stuf, function(a, b) return a.count > b.count end)
  local res = { path = list, stuf = stuf }
  if not currentbest then return res end
  return better(stuf, currentbest.stuf) and res or currentbest
end

local function findbest(fill, start, stop)
  local path = { stop }
  local currentbest
  local function halp()
  local p = path[#path]
    if p == start then
      currentbest = best(path, currentbest)
      return
    end
    local seen = fill[p]
    if not seen then return end
    local i = #path + 1
    for _, pp in pairs(seen.from) do
      path[i] = pp
      halp()
    end
    path[i] = nil
    return
  end
  halp()
  return currentbest.path
end 

function test2(str)
  local map = mappy(str)
  local start, stop = next(map.S), next(map.E)
  local fill = flood(map, start).seen
  local path = findbest(fill, start, stop)
  for _, p in ipairs(path) do
    map[p] = "o"
  end
  web.html(maphtml(map))
end

local function fillback(fill, p)
  local set, visited = { [p] = true }, { [p] = true }
  while next(set) do
    local p = next(set)
    set[p] = nil
    local seen = fill[p]
    for _, d in ipairs(dirs) do
      local from = seen.from[d]

local function shoot(p, dir, back)

``` prelude
function memo(key, new)
  local all = setmetatable({}, { __mode = "v" })
  return function(...)
    local k = key(...)
    local found = all[k]
    if found then return found end
    local v = new(...)
    all[k] = v
    return v
  end
end

Vec = {}
vec = memo(
  function(x, y) return x .. "," .. y end,
  function(x, y) return setmetatable({ x = x, y = y }, Vec) end)

function Vec.__add(a, b) return vec(a.x + b.x, a.y + b.y) end
function Vec.__sub(a, b) return vec(a.x - b.x, a.y - b.y) end
N, E, S, W = vec(0, -1), vec(1, 0), vec(0, 1), vec(-1, 0)
NW, NE, SE, SW = N + W, N + E, S + E, S + W
function Vec.__tostring(v) return v.x .. "," .. v.y end
dirs = { NW, NE, SE, SW, N, E, S, W }

Lazy = {}
function Lazy.__index(t, k)
  local list = {}
  t[k] = list
  return list
end

Queue = {}
Queue.__index = Queue
function Queue.new()
    local q = {
        highest = 0,
        lowest = 0,
        seen = {},
        lists = setmetatable({}, Lazy)
    }
    setmetatable(q, Queue)
    return q
end

function Queue.put(q, pos, cost, dir, from)
  local seen = q.seen[pos]
  if seen then
    if seen.cost == cost then
      seen.from[dir] = from
      seen.dirs = seen.dirs + 1
      return
    elseif seen.cost < cost then
      return
    end
    local prev = q.lists[seen.cost]
    local current = prev.next
    while current do
      if current.pos == pos then
        prev.next = current.next
        break
      end
      local next = current.next
      prev = current
      current = next
    end
  end
  q.seen[pos] = { cost = cost, from = dir and { [dir] = from } or {}, dirs = dir and 1 or 0 }
  q.lists[cost].next = { pos = pos, next = q.lists[cost].next }
  q.highest = math.max(q.highest, cost)
end

function Queue.get(q)
  local i = q.lowest
  for i = q.lowest, q.highest do
    local list = q.lists[i]
    if list.next then
      local entry = list.next
       list.next = entry.next
       local pos = entry.pos
       q.lowest = i
       return pos, i
    end
  end
end

function mappy(str)
  local lines = str:gmatch("[^\n]+")
  local map = {}
  local y = 0
  local w = 0
  for line in lines do
    y = y + 1
    local x = 0
    for c in line:gmatch(".") do
      x = x + 1
      local v = vec(x, y)
      if c == "#" then
        map[v] = c
      end
      local set = map[c] or {}
      set[v] = true
      map[c] = set
    end
    w = math.max(w, x)
  end
  map.size = vec(w, y)
  return map
end

function maphtml(map, f)
  local html = { '<pre><code>' }
  for y = 1, map.size.y do
    for x = 1, map.size.x do
      local p = vec(x, y)
      local c =
        (map.S[p] and 'S')
        or (map.E[p] and 'E')
        or map[p] or ' '
      html[#html + 1] = f and f(c, p) or c
    end
    html[#html + 1] = '\n'
  end
  return table.concat(html)
end

function neighbours(p, map)
  local i = 1
  local cost = 3
  return function()
    while i <= 8 do
      local dir = dirs[i]
      local np = p + dir
      if i == 5 then cost = 2 end
      i = i + 1
      if not map[np] then
        return np, cost, dir
      end
    end
    return nil
  end
end

function step(map, q)
  local p, cost = q:get()
  if not p then return false end
  for np, c, dir in neighbours(p, map) do
    q:put(np, cost + c, dir, p)
  end
  return true
end

function flood(map, start)
  local q = Queue.new()
  q:put(start, 0)
  local current = true
  while current do current = step(map, q, stop) end
  return q
end

function span(str, style)
  return '<span style="' .. style .. '">' .. str .. '</span>'
end

function heatstyle(n, max)
  local h = (1 - (n / max)) * 240
  return 'background-color: hsl(' .. h .. ', 100%, 50%)'
end

function heatcoloured(c, p, q)
  local seen = q.seen[p]
  return seen and span(c, heatstyle(seen.cost, q.highest)) or c
end
