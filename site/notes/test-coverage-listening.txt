``` run
filename("file1.lua")

ItemNumber_ = {}

function ItemNumber_.new(str)
  str = str:upper()
  str = str:match("^%s*(%a%a[-]%d+)%s*$")
  if not str then
    error({ message = "bad pattern" })
  end
  str = str:upper()
  if #str < 5 then
    error({ message = "few chars" })
  end
  if #str > 10 then
    error({ message = "many chars" })
  end
  return {
    tag = function()
      return "itemnumber"
    end,
    number = function()
      return str
    end
  } 
end
```

``` repl
runtests(
  function()
    assertEq(ItemNumber_.new("as-123456").number(), "AS-123456")
  end,
  function()
    local err = assertError(function() ItemNumber_.new("asd-123456") end)
    assertEq("bad pattern", err.message)
  end,
  function()
    local err = assertError(function() ItemNumber_.new("qw-1") end)
    assertEq("few chars", err.message)
  end,
  function()
    local err = assertError(function() ItemNumber_.new("ab-123456789") end)
    assertEq("many chars", err.message)
  end
)
```

``` prelude
local filenames = {}
function filename(name)
  local source = debug.getinfo(2).source
  filenames[source] = name
  filenames[name] = source
end

web.html([[<style>.covered { text-decoration: none; background-color: #d4fcbc; color: #000000; }</style>]])

local current = {}
local function trace(event, line)
  if not current then return end
  local file = filenames[debug.getinfo(2).source]
  if not file then return end
  local lines = current[file] or {}
  lines[line] = true
  current[file] = lines
end

local function linehtml(str, covered)
  local class = covered and [[ class="covered"]] or ""
  return "<span" .. class .. ">" .. str .. "</span><br>"
end

local function reportcoverage(t)
  local res = { "<details><summary>Code coverage report</summary><pre><code>" }
  for k, v in pairs(t) do
    res[#res + 1] = linehtml(k)
    local i = 0
    for line in filenames[k]:gmatch("[^\n]*") do
      i = i + 1
      res[#res + 1] = linehtml("  " .. line, v[i])
    end
  end
  res[#res + 1] = "</code></pre></details>"
  web.html(table.concat(res))
end

local function coveragestart()
  current = {}
  debug.sethook(trace, "l")
end

local escapechar = {
  ["<"] = "&lt;",
  [">"] = "&gt;",
  ['"'] = "&quot;",
  ["'"] =  "&apos;",
  ["&"] = "&amp;"
}

local function escape(str)
  local res, _ = (str or ""):gsub("[<>\"'&]", escapechar)
  return res
end

function runtests(...)
  coveragestart()
  local str = { '<p>' }
  local passes, failures = 0, 0
  for i, f in ipairs({...}) do
    local success, res = pcall(f)
    if success then
      passes = passes + 1
    else
      failures = failures + 1
      table.insert(str, 'test ' .. i .. ' failed: ')
      table.insert(str, escape(tostring(res)) .. '<br>')
    end
  end
  table.insert(str, '<span style="background-color:')
  if failures == 0 then
    table.insert(str, 'green;">this is a green bar ')
  else
    table.insert(str, 'red;">this is a red bar ')
  end
  table.insert(str,  '(' .. passes .. ' tests passed. ')
  table.insert(str,  failures .. ' tests failed)</span></p>')
  web.html(table.concat(str))
  reportcoverage(current)
end

function assertEq(e, a)
  if e ~= a then
    error("expected: " .. tostring(e).. ", actual: " .. tostring(a))
  end
end

function assertError(f)
  local success, res = pcall(f)
  if success then error("expected: error, got: success") end
  return res
end
```
