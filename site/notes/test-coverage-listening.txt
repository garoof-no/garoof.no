ItemNumber__ = {}

function ItemNumber__.valid(str)
  str = str:match("^%a%a[-]%d+$")
  if not str then
    return false
  end
  if #str < 5 then
    return false
  end
  if #str > 10 then
    return false
  end
  return true
end

function ItemNumber__.new(str)
  str = trim(str):upper()
  if not valid(str) then
    error("bad format")
  end
  return {
    type = function()
      return ItemNumber__
    end,
    number = function()
      return str
    end
  }
end
function ItemNumber__.maybe(str)
  str = trim(str):upper()
  str = str:match("^%a%a[-]%d+$")
  if not valid(str) then
    error("bad format")
  end
  return ItemNumber__.new(str)
end

ItemNumber = {}

function ItemNumber.valid(str)
  str = trim(str):upper()
  str = str:match("^%a%a[-]%d+$")
  if not str then
    return Result.bad("bad format")
  end
  if #str < 5 then
    return Result.bad("bad format")
  end
  if #str > 10 then
    return Result.bad("bad format")
  end
  return Result.good(str)
end

function ItemNumber.new(str)
  str = valid(str).orError()
  return {
    type = function()
      return ItemNumber__
    end,
    number = function()
      return str
    end
  }
end
function ItemNumber.maybe(str)
  return valid(str).map(ItemNumber.new)
end

local files = {}
function traced()
  local n = #files + 1
  local source = debug.getinfo(2).source
  files[source] = n
  files[n] = source
end

web.html([[<style>.covered { text-decoration: none; background-color: #d4fcbc; color: #000000; }</style>]])

local current = {}
local function trace(event, line)
  if not current then return end
  local file = files[debug.getinfo(2).source]
  if not file then return end
  local lines = current[file] or {}
  lines[line] = true
  current[file] = lines
end

local function linehtml(str, covered)
  local class = covered and [[ class="covered"]] or ""
  return "  <span" .. class .. ">" .. str .. "</span><br>"
end

local function blank(s)
  return not not s:match("^%s*$")
end

local function endline(s)
  return not not s:match("^%s*end%s*$")
end

local function reportcoverage(t)
  local res = { "<details><summary>Code coverage report</summary><pre><code>" }
  for n, v in pairs(t) do
    res[#res + 1] = "file " .. n .. ":"
    local prev = false
    local i = 0
    for line in files[n]:gmatch("[^\n]*") do
      i = i + 1
      local covered = v[i] or (prev and (blank(line) or endline(line)))
      prev = covered
      res[#res + 1] = linehtml(line, covered)
    end
  end
  res[#res + 1] = "</code></pre></details>"
  web.html(table.concat(res))
end

local function coveragestart()
  current = {}
  debug.sethook(trace, "l")
end

local escapechar = {
  ["<"] = "&lt;",
  [">"] = "&gt;",
  ['"'] = "&quot;",
  ["'"] =  "&apos;",
  ["&"] = "&amp;"
}

local function escape(str)
  local res, _ = (str or ""):gsub("[<>\"'&]", escapechar)
  return res
end

function runtests(...)
  coveragestart()
  local str = { '<p>' }
  local passes, failures = 0, 0
 
  for i, f in ipairs({...}) do
    local success, res = pcall(f)
    if success then
      passes = passes + 1
    else
      failures = failures + 1
      table.insert(str, 'test ' .. i .. ' failed: ')
      table.insert(str, escape(tostring(res)) .. '<br>')
    end
  end
  table.insert(str, '<span style="background-color:')
  if failures == 0 then
    table.insert(str, 'green;">this is a green bar ')
  else
    table.insert(str, 'red;">this is a red bar ')
  end
  table.insert(str,  '(' .. passes .. ' tests passed. ')
  table.insert(str,  failures .. ' tests failed)</span></p>')
  web.html(table.concat(str))
  reportcoverage(current)
end

function assertEqual(e, a)
  if e ~= a then
    error("expected: " .. tostring(e).. ", actual: " .. tostring(a))
  end
end

function assertError(f)
  local success, res = pcall(f)
  if success then error("expected: error, got: success") end
  return res
end

Result = {}
function Result.good(v)
  return {
    value = function() return v end,
    orError = function() return v end,
    map = function(f) return Result.ok(f(v)) end
  }
end

function Result.bad(m)
  local self
  self = {
    message = function() return m end,
    orError = function() error(m) end,
    map = function(f) return self end
  }
  return self
end

function trim(s)
  return s:match("^%s*(.-)%s*$")
end
